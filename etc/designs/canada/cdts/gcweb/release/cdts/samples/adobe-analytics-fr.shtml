document.write "refTop" élément script

Utiliser Adobe Analytics version 1 sur l'environnement staging.

document.write(wet.builder.refTop({
    "webAnalytics" : [{
        "environment" : "staging",
        "version" : 1
    }]
}));

Utiliser Adobe Analytics version 2 sur l'environnement production.

document.write(wet.builder.refTop({
    "webAnalytics" : [{
        "environment" : "production",
        "version" : 2
    }]
}));

document.write "refFooter" élément script

document.write(wet.builder.refFooter({
    "webAnalytics" : true
}));
